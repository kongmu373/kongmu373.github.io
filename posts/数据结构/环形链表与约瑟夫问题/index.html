Josephu 问题

n个人围着作，编号分别为1,2,… n ,约定从编号k开始报数(1<=k<=n) 从1开始报数，数到m的那个人出列，然后下一位又从1开始报数到m，以此类推，直到所有人都出列，求产生的出队编号序列 note: 循环链表

import java.util.ArrayList;
import java.util.List;

/**
 * 1. 先构建 一个 node, 创建first遍历，让其指向自身
 * 2. 后面每创建一个 node, 就把该节点指向第一个，形成环形链表
 */
public class Josephu {
    public static void main(String[] args) {
        Josephu josephu = new Josephu();
        josephu.init(5);
        josephu.show();
        josephu.count(1, 2, 5);
        System.out.println(josephu.getNoList());
    }

    private Node head = null;
    private Node helper = null;
    public Node startNode = null;
    private List<Integer> noList = new ArrayList<>();

    public List<Integer> getNoList() {
        return noList;
    }

    public void init(int nums) {
        if (!valid(nums)) {
            System.out.println("nums值不正确");
            return;
        }
        Node cur = head;
        for (int i = 1; i <= nums; i++) {
            Node node = new Node(i);
            if (i == 1) {
                head = node;
                head.next = head;
                cur = head;
            } else {
                cur.next = node;
                node.next = head;
                cur = node;
            }
        }
    }

    // show
    public void show() {
        if (isEmpty()) {
            System.out.println("链表为空");
            return;
        }

        Node cur = head;
        while (cur.next != head) {
            System.out.println(cur);
            cur = cur.next;
        }
        System.out.println(cur);
    }

    /**
     * @param start 从第几个开始
     * @param count 要数的数目
     * @param nums  最初有多少小孩
     */
    public void count(int start, int count, int nums) {
        if (nums == 1) {
            noList.add(startNode.no);
            return;
        }
        if (nums < 1) {
            return;
        }
        if (startNode == null || helper.next != startNode) {
            traverseToEnd(start);
        }

        for (int i = 0; i < count - 1; i++) {
            startNode = startNode.next;
            helper = helper.next;
        }
        noList.add(startNode.no);
        startNode = startNode.next;
        helper.next = startNode;
        count(startNode.no, count, nums - 1);
    }

    private void traverseToEnd(int start) {
        Node temp = head;
        while (true) {
            if (temp.no == start) {
                startNode = temp;
                break;
            }
            if (temp.next == head) {
                throw new RuntimeException("找不到该值");
            }
            temp = temp.next;
        }

        while (temp.next != startNode) {
            temp = temp.next;
        }
        helper = temp;
    }

    private boolean isEmpty() {
        return head == null;
    }

    private boolean valid(int nums) {
        return nums >= 1;
    }

    public static class Node {
        int no;
        Node next;

        public Node(int no) {
            this.no = no;
        }

        @Override
        public String toString() {
            return "Node{" +
                       "no=" + no +
                       '}';
        }
    }
}
